Google Sheets 實時雙向同步專用代碼

您可以在 Google 試算表內點選 擴充功能 > Apps Script，貼上下方代碼並部署為網頁應用程式。此代碼支援即時比對日期進行覆寫更新或刪除，解決重複與同步問題： 

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  var dateStr = data.date;
  
  // 1. 自動比對日期定位該列
  var rows = sheet.getDataRange().getValues();
  var targetRowIndex = -1;
  for (var i = 1; i < rows.length; i++) {
    var cellDate = rows[i][1];
    if (cellDate && (cellDate == dateStr || Utilities.formatDate(new Date(cellDate), "GMT+8", "yyyy/MM/dd") == dateStr)) {
      targetRowIndex = i + 1;
      break;
    }
  }
  
  // 2. 處理刪除要求
  if (data.action === "DELETE") {
    if (targetRowIndex > -1) {
      sheet.deleteRow(targetRowIndex);
      return ContentService.createTextOutput("Deleted");
    }
    return ContentService.createTextOutput("Not Found");
  }
  
  var keys = Object.keys(data).filter(function(k){ 
    return k !== 'date' && k !== 'room' && k !== 'roomName' && k !== 'action'; 
  });
  var headers = rows[0];
  
  // 3. 日期存在則「覆寫更新」，不存在則「新增行」
  if (targetRowIndex > -1) {
    keys.forEach(function(k) {
      var colIdx = headers.indexOf(k);
      if (colIdx > -1) {
        sheet.getRange(targetRowIndex, colIdx + 1).setValue(data[k]);
      }
    });
    sheet.getRange(targetRowIndex, 1).setValue(new Date()); // 修改戳記
    return ContentService.createTextOutput("Updated");
  } else {
    var newRow = new Array(headers.length).fill("");
    newRow[0] = new Date();
    newRow[1] = dateStr;
    keys.forEach(function(k) {
      var colIdx = headers.indexOf(k);
      if (colIdx > -1) newRow[colIdx] = data[k];
    });
    sheet.appendRow(newRow);
    return ContentService.createTextOutput("Inserted");
  }
}

💡 雙向同步技術： 前端不論是新增、點擊修改格子或刪除，都會呼叫此 API，試算表數據保持 100% 相同。
